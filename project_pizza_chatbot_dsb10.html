menu <- c(
  "Margherita Pizza", "Pepperoni Pizza",
  "BBQ Chicken Pizza", "Vegetarian Pizza",
  "Seafood Deluxe Pizza",
  "Coca-Cola", "Iced Lemon Tea",
  "Orange Juice", "Sparkling Water",
  "Mango Smoothie",
  "Chocolate Lava Cake", "Tiramisu",
  "Cheesecake", "Fruit Salad",
  "Ice Cream Sunday"
)

menu_lower <- tolower(menu)

price <- c(
  sample(15:20, 5),  
  sample(2:6, 5),    
  sample(7:12, 5)    
)

df_menu <- data.frame(MENU = menu, PRICE = price)

pizza <- function() {
  cat("\nWelcome to the Wonderful Pizza Shop!\n") 

  collect_order <- list()  
  
  while (TRUE) {
    
    cat("Order now! :\n") 
    cat("Type 'menu' to see the menu,\n") 
    cat("Type 'check' to view your basket and check the bill,\n") 
    cat("Type 'reorder' to clear your basket and start over,\n") 
    cat("Type 'stop' to leave the restaurant.\n") 
    
    order_item <- tolower(readline("What would you like to do? "))
    
    if (order_item %in% "menu") { 
      cat("\n----------- MENU -----------\n") 
      print(df_menu) 
      next 
    }

    if (order_item %in% "stop") { 
      break 
    }

    if (order_item %in% "check") { 
      if (length(collect_order) == 0) { 
        cat("\nYour basket is empty.\n") 
      } else { 
        original_menu_names <- sapply(names(collect_order), function(name) {
          menu[match(name, menu_lower)]
        })
        
        order_summary <- data.frame(
          Menu = unname(original_menu_names), 
          Quantity = unname(unlist(collect_order)),
          Unit_Price_Dollar = price[match(names(collect_order), menu_lower)],
          Total_Price_Dollar = unname(unlist(collect_order) * price[match(names(collect_order), menu_lower)]),
          stringsAsFactors = FALSE
        )
        cat("\nYour current order summary:\n") 
        print(order_summary) 
        
        total_order <- sum(order_summary$Total_Price_Dollar)
        cat("\nOrder Total: $", total_order, "\n", sep = "") 
        
        confirm_order <- tolower(readline("\nDo you confirm your order? (Y/N): "))

        if (confirm_order %in% "y") {
          cat("\nOrder confirmed! Thank you for your purchase.\n")
          return(order_summary)
        } else if (confirm_order %in% "n") {  
          cat("\nYou can continue ordering.\n")
          next 
        } else { 
          cat("Invalid input. Returning to menu.\n")
          next 
        }
      }
      next 
    }
    if (order_item %in% "reorder") {
      cat("\nYou chose to clear your basket and start over.\n") 
      collect_order <- list() 
      next 
    }
    if (order_item %in% menu_lower) { 
      order_quantity <- as.integer(readline(paste("How many", order_item, "would you like to order? "))) 

      if (is.na(order_quantity) || order_quantity <= 0) { 
        cat("Invalid quantity. Please enter only a positive integer!\n") 
        next 
      }

      if (order_item %in% names(collect_order)) { 
        confirm_add <- tolower(readline("You have already ordered this item. Do you want to add more? (Y/N): ")) 

        if (confirm_add %in% "y") { 
          collect_order[[order_item]] <- collect_order[[order_item]] + order_quantity 
          cat("Updated:", order_item, "-", collect_order[[order_item]], "in total.\n") 
        } else if (confirm_add %in% "n") { 
          next 
        } else { 
          cat("Invalid input. Skipping this item.\n") 
          next 
        }
      } else { 
        collect_order[[order_item]] <- order_quantity 
        cat("Added:", order_item, "->", order_quantity, "\n") 
      }
    } else { 
      cat("Invalid input. Please try again.\n") 
    }
  }

  if (length(collect_order) == 0) { 
    print("Your basket is currently empty. Thank you for visiting and using our service!") 
  } else { 
    print("If you return for a new visit, the items in your basket will no longer be there. We warmly invite you and look forward to welcoming you again!")
  }
}

order_summary <- pizza()
