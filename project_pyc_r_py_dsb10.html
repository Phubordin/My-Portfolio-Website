hammer_scissor_paper <- function() {
  
  choices <- c("hammer", "scissor", "paper")
  
  user_score <- 0
  computer_score <- 0

  results <- data.frame(
    Round = integer(),
    Winner = character(),
    User_Choice = character(),
    Computer_Choice = character(),
    stringsAsFactors = FALSE
  )

  cat("** Welcome to Hammer, Scissor, Paper Game! **\n")
  cat("Rules:\n")
  cat("- You must play 10 rounds to complete the game.\n")
  cat("- Type 'check' at any time to see the current scores and game status.\n")
  cat("- Only valid inputs are: 'hammer', 'scissor', 'paper' (case-insensitive).\n")
  cat("Let's begin!\n")

  for (round in 1:10) {
    cat("\n** Round", round, "**\n")

    computer_choice <- sample(choices, 1)

    valid_input <- FALSE

    while (!valid_input) {

      user_input <- tolower(gsub(" ", "",readline(paste0(
        "Your choice (hammer, scissor, paper): "
      ))))

      if (user_input == "check") {
        cat("\n** Current Game Status **\n")
        print(results)
        cat("\nUser Score:", user_score, "| Computer Score:", computer_score, "\n")
        cat("Returning to Round", round, "\n")
        next
      }

      if (!(user_input %in% choices)) {
        cat("Invalid input. Please type only 'hammer', 'scissor', or 'paper'. Try again.\n")
        next
      }

      valid_input <- TRUE
    }

    cat("You chose:", user_input, "| Computer chose:", computer_choice, "\n")

    if (user_input == computer_choice) {
      winner <- "Draw"
      cat("Result: It's a draw!\n")
    } else if (
      (user_input == "hammer" && computer_choice == "scissor") ||
      (user_input == "scissor" && computer_choice == "paper") ||
      (user_input == "paper" && computer_choice == "hammer")
    ) {
      winner <- "User"
      user_score <- user_score + 1
      cat("Result: You win this round!\n")
    } else {
      winner <- "Computer"
      computer_score <- computer_score + 1
      cat("Result: Computer wins this round!\n")
    }

    results <- rbind(
      results,
      data.frame(
        Round = round,
        Winner = winner,
        User_Choice = user_input,
        Computer_Choice = computer_choice,
        stringsAsFactors = FALSE
      )
    )
  }

  cat("\n** Final Results **\n")
  print(results)

  if (user_score > computer_score) {
    cat("\nOverall Winner: User with", user_score, "points!\n")
    cat("Computer scored", computer_score, "points.\n")
    cat("User won", user_score - computer_score, "more rounds than Computer.\n")
  } else if (computer_score > user_score) {
    cat("\nOverall Winner: Computer with", computer_score, "points!\n")
    cat("User scored", user_score, "points.\n")
    cat("Computer won", computer_score - user_score, "more rounds than User.\n")
  } else {
    cat("\nOverall Result: It's a draw with both scoring", user_score, "points!\n")
  }
}
