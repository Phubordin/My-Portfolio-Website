class Atm():    
  
  def __init__(self, name, bank, balance, statement):        
    self.name = name         
    self.bank = bank         
    self.balance = balance   
    self.statement = statement     
  
  def __str__(self):
    text = "You should type (my_atm.Attribute | my_atm.Methods(__, __) to view info." 
    return text 
  
  def withdraw(self, num, hint):
    self.balance -= num  
    self.statement["#Hints"] += [hint]       
    self.statement["$Cash"] += [f"- {num}"]   
    self.statement["Transactions"] += ["Withdraw"] 
    if self.balance < 0:  
      return "You don't have enough money"    
    else:
      text = f"""WithDraw: - {num} | * {hint}"""  
      return text 
  
  def paybills(self, num, hint):
    self.balance -= num  
    self.statement["#Hints"] += [hint]       
    self.statement["$Cash"] += [f"- {num}"]   
    self.statement["Transactions"] += ["Pay Bills"] 
    if self.balance < 0:  
      return "You don't have enough money"    
    else:
      text = f"""Pay Bills: - {num} | * {hint}"""  
      return text 
  
  def deposit(self, num, hint):
    self.balance += num  
    self.statement["#Hints"] += [hint]       
    self.statement["$Cash"] += [f"+ {num}"]   
    self.statement["Transactions"] += ["Pay Bills"] 
    text = f"""Deposit : + {num} | * {hint}""" 
    return text
